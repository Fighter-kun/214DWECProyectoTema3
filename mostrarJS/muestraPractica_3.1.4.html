4. Crea una aplicación que pida palabras al usuario hasta que pulse el botón cancelar:


                             /*4. Crea una aplicación que pida palabras al usuario hasta que pulse el botón cancelar. 
                            Después aparecerá un menú de opciones: 
                               1) Ver primera palabra insertada 
                               2) Ver última palabra insertada 
                               3) Ver todas las palabras 
                               4) Ver la palabra más larga 
                               5) Ver la palabra más corta 
                               6) Número de palabras insertadas 
                               7) Buscar una palabra 
                               8) Borrar duplicados 
                               9) Salir 
                                */
                         var aCadena = [];
                         var entrada = "";
                         var bSalida = true;
                         do {
                            entrada = prompt("Introduce palabras: ");
                            if (entrada != null) {
                               aCadena.push(entrada);
                            }
                         } while (entrada != null);

                         do {
                            const opcion = parseInt(prompt(`Menú de opciones:
                           1) Ver primera palabra insertada
                           2) Ver última palabra insertada
                           3) Ver todas las palabras
                           4) Ver la palabra más larga
                           5) Ver la palabra más corta
                           6) Número de palabras insertadas
                           7) Buscar una palabra
                           8) Borrar duplicados
                           9) Salir
                           Ingrese el número de la opción:`));
                         switch (opcion) {
                            case 1:
                               alert(aCadena[0]);
                               break;
                            case 2:
                               alert(aCadena[aCadena.length-1]);
                               break;
                            case 3:
                               for (let i = 0; i < aCadena.length; i++) {
                                  if (aCadena[i] != undefined) {
                                     alert(`La palabra en la posición ${i} es: ${aCadena[i]}`);
                                  }
                               }
                               break;
                            case 4:
                               var palabraMayor = aCadena[0];
                               for (let i = 1; i < aCadena.length; i++) {
                                  if (aCadena[i] != undefined) {
                                     if (aCadena[i].length > palabraMayor.length) {
                                           palabraMayor = aCadena[i];
                                     }
                                  }
                               }
                               alert("La palabra más larga es "+palabraMayor);
                               break;
                            case 5:
                               var palabraMenor = aCadena[0];
                               for (let i = 1; i < aCadena.length; i++) {
                                  if (aCadena[i] != undefined) {
                                     if (aCadena[i].length < palabraMenor.length) {
                                           palabraMenor = aCadena[i];
                                     }
                                  }
                               }
                               alert("La palabra más corta es "+palabraMenor);
                               break;
                            case 6:
                               alert("Hay "+aCadena.length+" palabras.");
                               break;
                            case 7:
                               var palabra = prompt("Introduce palabra a buscar: ");
                               if (aCadena.includes(palabra)) {
                                  alert("Se encontro la palabra.");
                               } else {
                                  alert("No se encontro la palabra.");
                               }
                               break;
                            case 8:
                               var aCadenaSinDuplicados =  [...new Set(aCadena)];
                               alert("Array sin duplicados "+aCadenaSinDuplicados.join(','));
                               break;
                            case 9:
                               alert("Hasta luego.")
                               bSalida = false;
                               break;
                            default:
                               alert("ERROR EN LA ENTRADA");
                               break;
                         }
                         } while (bSalida);
