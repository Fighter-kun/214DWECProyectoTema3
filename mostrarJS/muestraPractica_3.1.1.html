                            /*1. Crea un script que pida una frase al usuario y se la muestre con la primera letra de cada palabra en mayúsculas.*/
                            var fraseUsuario = prompt("Introduce un frase: ");
                            //Formamos un array sustituyendo los espacion por comas.
                            var aFraseUsuario = fraseUsuario.split(" ");
                            /*
                            Ahora con '.map' cogemos cada elemento del array, con '.charAt(0)' devuelve el primer carácter, con 'toUpperCase()' ponemos esa letra en mayúscula y
                            con '.slice' devolvemos el resto de la palabra a partir del segundo carácter.
                            */
                            var aFraseUsuarioLetraMayuscula = aFraseUsuario.map(aFraseUsuario => aFraseUsuario.charAt(0).toUpperCase() + aFraseUsuario.slice(1).toLowerCase());
                            //Separamos el array 'aFraseUsuarioLetraMayuscula' y sustituimos las comas por espacios.
                            var salida = aFraseUsuarioLetraMayuscula.join(" ");
                            //Si el usuario escribe una frase se devuelve 'Salida' si devuelve el contenido del else.
                            if(fraseUsuario) {
                               document.write(salida);
                            } else {
                               document.write("No ingreso ninguna frase.");
                            }
