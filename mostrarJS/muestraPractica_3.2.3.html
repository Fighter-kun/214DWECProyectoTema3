3. Crear una función para comprobar hasta que punto la función Math.random() es realmente aleatorio:


                            /*
                            3. Crear una función para comprobar hasta que punto la función Math.random() es realmente aleatoria. 
                            La función recibirá un parámetro que por defecto será 10000. Ese, será el número de veces que generaremos números aleatorios del 1 al 10. 
                            Aunque el ejercicio se puede resolver con arrays, utiliza un mapa donde las claves serán los números del 1 al 10 y los valores, el número de veces que aparece.  
                            */
                           //Función para contar repeticiones de palabras
                           function contarRepeticionesNumerosAleatorios() {
                              var mapNumeros = new Map(); //Declaración del map

                              //Inicializamos las claves y valores 
                              mapNumeros.set(1, 0);
                              mapNumeros.set(2, 0);
                              mapNumeros.set(3, 0);
                              mapNumeros.set(4, 0);
                              mapNumeros.set(5, 0);
                              mapNumeros.set(6, 0);
                              mapNumeros.set(7, 0);
                              mapNumeros.set(8, 0);
                              mapNumeros.set(9, 0);
                              mapNumeros.set(10, 0);

                              for (let i = 0; i < 10000; i++) {
                                var numeroAleatorio = Math.floor(Math.random() * 10) + 1; //Calculamos el número aleatorio entre 1 y 10.
                                if (mapNumeros.has(numeroAleatorio)) { //Comprobamos si el número ingresado esta almacenado
                                  mapNumeros.set(numeroAleatorio, mapNumeros.get(numeroAleatorio) + 1); // Si el número existe, sumamos 1 al valor de ese número
                                } else {
                                  mapNumeros.set(numeroAleatorio, 1); //Si no esta almacenado, lo guardamos y lo inicializamos con valor 1
                                }
                              }

                              return mapNumeros; //Devolvemos el mapa de números 
                            }

                            // Función para mostrar los resultados
                            function mostrarResultados(mapNumeros) {
                              mapNumeros.forEach((repeticiones,numero) => {
                                document.write("El "+numero+" se a repetido: "+repeticiones+" veces.
");
                              });
                            }

                            const mapaResultados = contarRepeticionesNumerosAleatorios(); //Llamamos a la función para iniciar el ejercicio.
                            mostrarResultados(mapaResultados); //Llamamos a la función para mostrar los resultados.
