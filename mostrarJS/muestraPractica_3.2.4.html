                            /*
                            4. Crea una función llamada invertirMapa. 
                            La función devolverá un nuevo mapa donde las claves serán los valores del mapa original y los valores del nuevo mapa 
                            serán arrays que contendrán las claves donde ese valor se repetía. 
                            Ejemplo:   
                              Mapa Original: 
                                -Clave: 1111BCD Valor: Seat 
                                -Clave: 2222DEF Valor: Audi 
                                -Clave: 3333FGH Valor: Mercedes 
                                -Clave: 4444HIJ Valor: Seat 
                                -Clave: 5555JKL Valor: Mercedes 
                                -Clave: 6666LMN Valor: Seat 
                              Mapa Invertido: 
                                -Clave: Seat  Valor: [1111BCD, 4444HIJ, 6666LMN] 
                                -Clave: Audi  Valor: [2222DEF] 
                                -Clave: Mercedes Valor: [3333FGH, 5555JKL]  
                            */
                            var mapMarcasCoche = new Map(); //Declaro el mapa

                            //Inicializo las claves y valores 
                            mapMarcasCoche.set('1111BCD', 'Seat');
                            mapMarcasCoche.set('2222DEF', 'Audi');
                            mapMarcasCoche.set('3333FGH', 'Mercedes');
                            mapMarcasCoche.set('4444HIJ', 'Seat');
                            mapMarcasCoche.set('5555JKL', 'Mercedes');
                            mapMarcasCoche.set('6666LMN', 'Seat');

                           function invertirMapa(mapMarcasCoche) {
                            var mapInvertido = new Map();
                            mapMarcasCoche.forEach((valor, clave) => {
                              if (!mapInvertido.has(valor)) {
                                mapInvertido.set(valor, [clave]);
                              } else {
                                mapInvertido.get(valor).push(clave);
                              }
                            });
                            return mapInvertido;
                           }

                           // Función para mostrar los resultados
                           function mostrarResultados(mapInvertido) {
                            mapInvertido.forEach((valor,clave) => {
                              document.write("< tr>< td>"+valor+"< /td>< td>"+clave+"< /td>< /tr>");
                              });
                            }
